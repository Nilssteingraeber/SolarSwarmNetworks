package "custom_interfaces.msg" {
  class Header {
    nid: str
    frame_id: str
    time: Time
  }
  
  class Time {
    sec: int
  }  

  class RobotBattery {
    header : Header
    data : float
  }

  class RobotCpu {
    header : Header
    data : float
  }

  class RobotActivity {
    header : Header
    activity : str
  }

  class RobotPoint {
    header : Header
    x : float
    y : float
    z : float
  }

  class RobotQuaternion {
    header : Header
    x : float
    y : float
    z : float
    w : float
  }

  class RobotMisc {
    header : Header
    ipv4 : str
    ipv6 : str
    mac : str
  }

  class NeighborList {
    header : Header
    neighbors : list
    indicators : list
  }

  Header --> Time
  RobotBattery --> Header
  RobotCpu --> Header
  RobotActivity --> Header
  RobotPoint --> Header
  RobotQuaternion --> Header
  RobotMisc --> Header
  NeighborList --> Header
}

package "custom_interfaces.srv" {
  class SetRobotActivity_Request {
    activity : str
    details : str
  }

  class SetRobotActivity_Response {
    msg : str
  }

  class RobotServiceInfo_Request {
    service : str
  }

  class RobotServiceInfo_Response {
    msg : str
    services : list
  }

  class RobotInterfaceInfo_Request {
    interface : str
  }

  class RobotInterfaceInfo_Response {
    definition : str
    interfaces : list
  }
}

SetRobotActivity_Request --- SetRobotActivity_Response
RobotServiceInfo_Request --- RobotServiceInfo_Response
RobotInterfaceInfo_Request --- RobotInterfaceInfo_Response
